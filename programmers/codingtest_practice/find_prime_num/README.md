def solution(n):
    num=set(range(2,n+1))
    for i in range(2,n+1):
        if i in num:
            num-=set(range(2*i,n+1,i))
    return len(num)

    if n == 1 : return count
    for i in range(2, n+1) :
        b = True

                break

    return count

def solution(n):
    l = [i for i in range(2,n)]
    for i in range(2,int(n**1/2)) :
        for j in range(0,n,i) :
            if j == i : continue;
            if j in l :
                l.remove(j)
    return len(l)
